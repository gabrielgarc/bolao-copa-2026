using CopaPixel.API.DTOs;
using CopaPixel.API.Models;
using CopaPixel.API.Repositories;

namespace CopaPixel.API.Services
{
    public class PredictionService
    {
        private readonly IRepository<Prediction> _predictionRepo;
        private readonly IRepository<Match> _matchRepo;
        private const string CURRENT_USER_ID = "u-current"; 

        public PredictionService(IRepository<Prediction> predictionRepo, IRepository<Match> matchRepo)
        {
            _predictionRepo = predictionRepo;
            _matchRepo = matchRepo;
        }

        public async Task<Dictionary<string, PredictionDto>> GetUserPredictionsAsync()
        {
            var preds = await _predictionRepo.FindAsync(p => p.UserId == CURRENT_USER_ID);

            var dict = new Dictionary<string, PredictionDto>();
            foreach(var p in preds)
            {
                dict[p.MatchId] = new PredictionDto(p.MatchId, p.HomeScore.ToString(), p.AwayScore.ToString());
            }
            return dict;
        }

        public async Task<(bool Success, string Message)> SavePredictionAsync(CreatePredictionDto dto)
        {
            if (!int.TryParse(dto.HomeScore, out int home) || !int.TryParse(dto.AwayScore, out int away))
            {
                return (false, "Placares inválidos");
            }

            var match = await _matchRepo.GetByIdAsync(dto.MatchId);
            if (match == null) return (false, "Jogo não encontrado");
            
            if (match.IsLocked || match.RealHomeScore.HasValue) 
                return (false, "Este jogo já está fechado para palpites");

            var existing = await _predictionRepo.FindOneAsync(p => p.UserId == CURRENT_USER_ID && p.MatchId == dto.MatchId);

            if (existing != null)
            {
                existing.HomeScore = home;
                existing.AwayScore = away;
                await _predictionRepo.UpdateAsync(existing.Id, existing);
            }
            else
            {
                await _predictionRepo.CreateAsync(new Prediction
                {
                    Id = MongoDB.Bson.ObjectId.GenerateNewId().ToString(),
                    UserId = CURRENT_USER_ID,
                    MatchId = dto.MatchId,
                    HomeScore = home,
                    AwayScore = away
                });
            }

            return (true, "Sucesso");
        }
    }
}
