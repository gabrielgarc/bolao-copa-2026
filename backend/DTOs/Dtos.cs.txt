namespace CopaPixel.API.DTOs
{
    // DTO para enviar times pro frontend
    public record TeamDto(
        string Id, 
        string Name, 
        string Code, 
        string FlagType, 
        List<string> Colors, 
        string TextColor
    );

    // DTO para enviar partidas pro frontend
    public record MatchDto(
        string Id,
        TeamDto HomeTeam,
        TeamDto AwayTeam,
        string Date,
        string Time,
        string Group,
        string Stadium,
        string Stage,
        int? RealHomeScore,
        int? RealAwayScore,
        bool IsLocked,
        string? PlaceholderLabel
    );

    // DTO para receber palpites
    public record CreatePredictionDto(
        string MatchId,
        string HomeScore, // Recebemos como string pra facilitar parsing do input html
        string AwayScore
    );

    // DTO para enviar palpites salvos
    public record PredictionDto(
        string MatchId,
        string HomeScore,
        string AwayScore
    );

    // DTO para o Ranking
    public record RankingDto(
        string Id,
        string Name,
        int Points,
        string Avatar
    );

    // DTO Usuário Atual
    public record UserDto(
        string Id,
        string Name,
        int Rank,
        int TotalPoints
    );
}
