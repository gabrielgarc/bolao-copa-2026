using CopaPixel.API.DTOs;
using CopaPixel.API.Services;
using Microsoft.AspNetCore.Mvc;

namespace CopaPixel.API.Controllers
{
    [ApiController]
    [Route("api/[controller]")]
    public class MatchesController : ControllerBase
    {
        private readonly MatchService _service;

        public MatchesController(MatchService service)
        {
            _service = service;
        }

        [HttpGet]
        public async Task<ActionResult<List<MatchDto>>> GetMatches()
        {
            var matches = await _service.GetAllMatchesAsync();
            return Ok(matches);
        }
    }
}
